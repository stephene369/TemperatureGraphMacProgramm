# Installez dmgbuild
pip install dmgbuild

# Créez un fichier de configuration pour le DMG
cat > dmg_settings.py << EOL
# -*- coding: utf-8 -*-
from __future__ import unicode_literals

import biplist
import os.path

# Chemins des fichiers
application = defines.get('app', 'dist/TemperatureGraph.app')
appname = os.path.basename(application)

# Nom du volume DMG
volname = "TemperatureGraph Installer"

# Format du DMG
format = defines.get('format', 'UDBZ')
size = defines.get('size', None)
files = [application]

# Position des éléments
icon_locations = {
    appname: (140, 120),
    'Applications': (500, 120)
}

# Lien symbolique vers /Applications
symlinks = {'Applications': '/Applications'}

# Options d'affichage
show_status_bar = False
show_tab_view = False
show_toolbar = False
show_pathbar = False
show_sidebar = False
window_rect = ((100, 100), (640, 280))
EOL

# Créez le DMG
dmgbuild -s dmg_settings.py "TemperatureGraph" "TemperatureGraph.dmg"
